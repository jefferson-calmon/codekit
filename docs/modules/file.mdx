    <Tab title="getImageDataURLFromFile">
        Lê um `File` de input e devolve a data URL, lançando erro se a leitura falhar.
        ```ts
        getImageDataURLFromFile(image: File): Promise<string>
        ```
        ```tsx
        function onFileChange(event: React.ChangeEvent<HTMLInputElement>) {
            const file = event.target.files?.[0];
            if (!file) return;

            getImageDataURLFromFile(file).then(setPreviewUrl);
        }
        ```
